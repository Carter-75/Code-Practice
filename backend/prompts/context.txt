function requestLogger(req, res, next) {
  const timestamp = new Date().toISOString();
  const method = req.method;
  const url = req.url;

  // Record start time on the request object for later latency calculation
  req.start = new Date();

  // Log in the exact required format
  console.log(`[${timestamp}] ${method} ${url}`);

  next();   // pass control to the next middleware/handler
}

async function authenticate(req, res, next) {
  // Extract token from Authorization header
  const authHeader = req.get('Authorization');
  const token = authHeader && authHeader.startsWith('Bearer ') 
                ? authHeader.substring(7) 
                : null;

  // (a)(b) No token or doesn't start with Bearer
  if (!token) {
    return res.status(401).json({ error: "Authentication required" });
  }

  try {
    // (c) Verify the JWT token (assume verifyJwt is already imported)
    const user = await verifyJwt(token);

    // (d) Attach user to request and continue
    req.user = user;
    next();
  } catch (err) {
    // (e) Token invalid or verification failed
    res.status(401).json({ error: "Unauthorized" });
  }
}

function requireRole(roles) {
  return function(req, res, next) {
    // Check if req.user.role is in the allowed roles array
    if (roles.includes(req.user.role)) {
      next();                    // role is allowed → continue
    } else {
      // Role not allowed
      res.status(403).json({ error: "Forbidden" });
    }
  };
}



async function authenticate(req, res, next) {
  const email = req.body.email;
  const password = req.body.password;

  // Find user by email
  const user = users.find(u => u.email === email);

  if (!user) {
    return res.status(401).json({ error: "Invalid email or password" });
  }

  // Verify password using bcrypt
  const isMatch = await bcrypt.compare(password, user.hashedPassword);

  if (!isMatch) {
    return res.status(401).json({ error: "Invalid email or password" });
  }

  // Success: store user info in session
  req.session.user = {
    id: user.id,
    email: user.email
  };

  next();
}

function requireAuth(req, res, next) {
  if (req.session && req.session.user) {
    return next();
  }

  // Not authenticated
  res.status(401).json({ error: "Unauthorized" });
}

// LocalStrategy configuration
passport.use(new LocalStrategy(
  { usernameField: 'email' },
  async (email, password, done) => {
    const user = users.find(u => u.email === email);
    if (!user) {
      return done(null, false);
    }

    const isMatch = await bcrypt.compare(password, user.hashedPassword);
    if (!isMatch) {
      return done(null, false);
    }

    return done(null, user);
  }
));

// serializeUser
passport.serializeUser((user, done) => {
  done(null, user.id);
});

// deserializeUser
passport.deserializeUser((id, done) => {
  const user = users.find(u => u.id === id);
  done(null, user);
});

// DELETE a course by id
app.delete('/api/courses/:id', isAuthenticated, (req, res) => {
  const id = parseInt(req.params.id);
  const index = courses.findIndex(c => c.id === id);
  if (index === -1) return res.status(404).json({ error: "Course not found" });

  // Remove this course from all students
  students.forEach(student => {
    student.courses = student.courses.filter(courseId => courseId !== id);
  });

  courses.splice(index, 1);
  res.json({ message: "Course deleted successfully" });
});

// ENROLL student X in course Y
app.post('/api/courses/:courseId/students/:studentId', isAuthenticated, (req, res) => {
  const courseId = parseInt(req.params.courseId);
  const studentId = parseInt(req.params.studentId);

  const course = courses.find(c => c.id === courseId);
  const student = students.find(s => s.id === studentId);

  if (!course || !student) {
    return res.status(404).json({ error: "Course or student not found" });
  }

  if (course.students.includes(studentId)) {
    return res.status(400).json({ error: "Student already enrolled in this course" });
  }

  course.students.push(studentId);
  student.courses.push(courseId);

  res.json({ 
    message: `Student ${studentId} enrolled in course ${courseId}`,
    course,
    student 
  });
});

// REMOVE student X from course Y

// 3. CREATE a new order
router.post('/orders', isAuthenticated, (req, res) => {
  const { products: orderItems, total } = req.body;

  if (!orderItems || !Array.isArray(orderItems) || orderItems.length === 0) {
    return res.status(400).json({ error: "Products array is required" });
  }

  // Check stock for every item and decrement
  for (let item of orderItems) {
    const product = products.find(p => p.id === item.productId);
    if (!product) {
      return res.status(400).json({ error: `Product ${item.productId} not found` });
    }
    if (product.stock < item.qty) {
      return res.status(400).json({ error: `Insufficient stock for product ${item.productId}` });
    }
    // Decrement stock
    product.stock -= item.qty;
  }

  const newOrder = {
    id: v4(),                    // UUID
    userId: req.user.id,         // from Passport
    products: orderItems,
    total: total,
    status: "pending"
  };

  orders.push(newOrder);
  res.status(201).json(newOrder);
});

});

// 5. GET single order by orderId (only if owned by user)
router.get('/orders/:orderId', isAuthenticated, (req, res) => {
  const order = orders.find(o => o.id === req.params.orderId);
  if (!order) {
    return res.status(404).json({ error: "Order not found" });
  }
  if (order.userId !== req.user.id) {
    return res.status(403).json({ error: "Forbidden" });
  }
  res.json(order);
});

// 6. DELETE a single order by orderId (only if owned by user)
router.delete('/orders/:orderId', isAuthenticated, (req, res) => {
  const index = orders.findIndex(o => o.id === req.params.orderId);
  if (index === -1) {
    return res.status(404).json({ error: "Order not found" });
  }
  if (orders[index].userId !== req.user.id) {
    return res.status(403).json({ error: "Forbidden" });
  }

  orders.splice(index, 1);
  res.status(204).end();   // No content
});

// 7. CANCEL a single order by orderId (only if owned by user)
router.patch('/orders/:orderId/cancel', isAuthenticated, (req, res) => {   // or use .put if you prefer
  const order = orders.find(o => o.id === req.params.orderId);
  if (!order) {
    return res.status(404).json({ error: "Order not found" });
  }
  if (order.userId !== req.user.id) {
    return res.status(403).json({ error: "Forbidden" });
  }
  if (order.status === "cancelled") {
    return res.status(400).json({ error: "Order is already cancelled" });
  }

  order.status = "cancelled";
  res.json(order);
});
